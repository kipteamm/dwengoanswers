# Useless
belgium_latitude = (49.5, 51.5)
belgium_longitude = (2.5, 6.4)

def is_in_belgium(row):
    latitude = float(row['breedtegraad'])
    longitude = float(row['lengtegraad'])
     
    return (
        belgium_latitude[0] <= latitude <= belgium_latitude[1] and
        belgium_longitude[0] <= longitude <= belgium_longitude[1]
    )

csv['IsInBelgium'] = csv.apply(lambda row: is_in_belgium(row), axis=1)

belgium_stations = csv[csv['IsInBelgium']]

x = belgium_stations['breedtegraad']
y = belgium_stations['lengtegraad']

# Useful
plt.figure(figsize=(10, 10))

plt.axis(xmin=49.5, xmax=51.5, ymin=2.5, ymax=6.4)
plt.xticks(np.arange(49.5, 51.5, step=1))
plt.yticks(np.arange(2.5, 6.5, step=1))

plt.scatter(x, y)

plt.title('Treinstations Belgi')
plt.ylabel('Lengtegraad')
plt.xlabel('Breedtegraad')

plt.show()
